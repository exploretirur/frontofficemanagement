📋 Apps Script Setup — One Time Only
1
Create a Google Sheet
Go to sheets.google.com → create a new sheet → name it anything (e.g. "EduTrack Data").
2

3
Deploy as Web App
Deploy → New Deployment → Web App. Set "Execute as" → Me. Set "Who has access" → Anyone. Click Deploy. Copy the Web App URL.

Apps Script Code (copy all)
⚠️ User accounts are now synced to Google Sheets via a Users sheet. Passwords are stored in plain text in that sheet — keep your spreadsheet private and share it only with trusted admins. The existing Apps Script below handles everything automatically; no changes needed.
// EduTrack Apps Script — paste ALL of this, replacing any existing code // After pasting: Deploy → New Deployment → Web App // Execute as: Me | Who has access: Anyone function doGet(e) { var params = e && e.parameter ? e.parameter : {}; var action = params.action || ''; return handleRequest(action, params); } function doPost(e) { var data = {}; try { data = JSON.parse(e.postData.contents); } catch(err) { data = {}; } var action = data.action || ''; return handleRequest(action, data); } function handleRequest(action, data) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var result; try { // ── PING ────────────────────────────────────────────────── if (action === 'ping') { result = { success: true, message: 'Connected!' }; // ── READ SHEET ──────────────────────────────────────────── } else if (action === 'getData') { var sheetName = data.sheet; var sh = ss.getSheetByName(sheetName); if (!sh || sh.getLastRow() < 1) { result = { success: true, data: [] }; } else { var vals = sh.getDataRange().getValues(); var headers = vals[0].map(String); var rows = vals.slice(1).filter(function(r) { return String(r[0]).trim() !== ''; }); result = { success: true, data: rows.map(function(r) { var obj = {}; headers.forEach(function(h, i) { obj[h] = (r[i] === null || r[i] === undefined) ? '' : String(r[i]); }); return obj; }) }; } // ── OVERWRITE ENTIRE SHEET ──────────────────────────────── } else if (action === 'saveAll') { var sheetName2 = String(data.sheet || ''); var headers2 = data.headers; // array of strings var rows2 = data.rows; // array of objects if (!sheetName2 || !Array.isArray(headers2)) { result = { success: false, error: 'saveAll: missing sheet or headers' }; } else { var sh2 = ss.getSheetByName(sheetName2) || ss.insertSheet(sheetName2); sh2.clearContents(); sh2.appendRow(headers2); if (Array.isArray(rows2) && rows2.length > 0) { var matrix = rows2.map(function(r) { return headers2.map(function(h) { var v = r[h]; return (v === null || v === undefined) ? '' : v; }); }); sh2.getRange(2, 1, matrix.length, headers2.length).setValues(matrix); } result = { success: true, saved: Array.isArray(rows2) ? rows2.length : 0 }; } // ── APPEND ONE ROW ──────────────────────────────────────── } else if (action === 'appendRow') { var sheetName3 = String(data.sheet || ''); var headers3 = data.headers; var row3 = data.row; if (!sheetName3 || !Array.isArray(headers3) || !row3) { result = { success: false, error: 'appendRow: missing sheet, headers, or row' }; } else { var sh3 = ss.getSheetByName(sheetName3) || ss.insertSheet(sheetName3); if (sh3.getLastRow() === 0) sh3.appendRow(headers3); sh3.appendRow(headers3.map(function(h) { var v = row3[h]; return (v === null || v === undefined) ? '' : v; })); result = { success: true }; } } else { result = { success: false, error: 'Unknown action: ' + action }; } } catch(err) { result = { success: false, error: err.toString() }; } return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); }
