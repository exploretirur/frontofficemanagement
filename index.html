🔗 Google Sheets Integration
ℹ️ Deploy a Google Apps Script as a Web App and paste the URL below.
👥 Sync User Accounts to Google Sheets

Push all user accounts (teachers, office staff) to a Users sheet in your connected Google Sheet. This keeps login credentials backed up in the cloud.

📊 Offline Mode — Export to Excel

Download all data as an Excel (.xlsx) file for offline backup or use without internet. All modules: Students, Payments, Attendance, Marks, Salary.

💡 Offline tip: Save this HTML file to your computer. It works completely offline using your browser — no internet needed. Export to Excel regularly for backup. Re-import Excel to restore data on any device.
🔄 Sync Data from Google Sheets

Pull existing student, payment and attendance data from your connected Google Sheet into this app. This overwrites local data with whatever is in the sheet.

⚠️ This will replace all local data with data from your Google Sheet. Make sure the Web App URL is saved first.
📖 Apps Script Setup

Step 1: Create a Google Sheet. Copy the Spreadsheet ID from the URL.

Step 2: Go to Extensions → Apps Script, paste the code below, save.

Step 3: Deploy → New Deployment → Web App. Set access to "Anyone". Copy the URL.

Step 4: Paste the URL in the field above and save.

function doPost(e){ try{ const data=JSON.parse(e.postData.contents); const ss=SpreadsheetApp.getActiveSpreadsheet(); if(data.action==='addStudent'){ const sh=getSheet(ss,'Students'); if(sh.getLastRow()===0) sh.appendRow(['Roll No','First Name','Last Name','DOB','Gender','Mobile','Email','Address','Course','Batch','Admission Date','Duration','Total Fees','Concession','Guardian','Guardian Mobile','Qualification','ID Type','Remarks','Status']); const roll='ST'+String(sh.getLastRow()).padStart(4,'0'); sh.appendRow([roll,data.fname,data.lname,data.dob,data.gender,data.mobile,data.email,data.address,data.course,data.batch,data.admDate,data.duration,data.totalFees,data.concession,data.guardian,data.guardianMobile,data.qual,data.idType,data.remarks,'Active']); return out({success:true,rollNo:roll}); } if(data.action==='addPayment'){ const sh=getSheet(ss,'Payments'); if(sh.getLastRow()===0) sh.appendRow(['Receipt No','Date','Roll No','Student','Course','Amount','Mode','Reference','Remarks']); const r='RCP'+String(Date.now()).slice(-6); sh.appendRow([r,data.date,data.rollNo,data.studentName,data.course,data.amount,data.mode,data.reference,data.remarks]); return out({success:true,receiptNo:r}); } if(data.action==='addAttendance'){ const sh=getSheet(ss,'Attendance'); if(sh.getLastRow()===0) sh.appendRow(['Date','Roll No','Student','Course','Batch','Status','Topic','Software']); data.records.forEach(r=>sh.appendRow([data.date,r.roll,r.name,data.course,data.batch,r.status,r.topic||'',r.software||''])); return out({success:true}); } if(data.action==='getData'){ const sh=ss.getSheetByName(data.sheet); if(!sh) return out({success:true,data:[]}); const vals=sh.getDataRange().getValues(); const headers=vals[0]; const rows=vals.slice(1); return out({success:true,data:rows.map(r=>Object.fromEntries(headers.map((h,i)=>[h,r[i]])))}); } if(data.action==='syncUsers'){ const sh=getSheet(ss,'Users'); sh.clearContents(); sh.appendRow(['Name','Username','Role','Department','Mobile','Courses Assigned','Synced At']); data.users.forEach(u=>sh.appendRow([u.name,u.username,u.role,u.dept||'',u.mobile||'',u.courses||'',new Date().toISOString()])); return out({success:true}); } if(data.action==='addSalary'){ const sh=getSheet(ss,'Salary'); if(sh.getLastRow()===0) sh.appendRow(['ID','Month','Staff','Basic','WorkDays','PresentDays','LeaveDays','OTHours','OTDays','OTAmount','Advance','Deductions','Bonus','GrossPay','NetPay','Remarks']); sh.appendRow([data.id,data.month,data.staffName,data.basic,data.workDays,data.presentDays,data.leaveDays,data.otHours,data.otDays,data.otAmount,data.advance,data.deductions,data.bonus,data.grossPay,data.netPay,data.remarks||'']); return out({success:true}); } if(data.action==='addMarks'){ const sh=getSheet(ss,'Marks'); if(sh.getLastRow()===0) sh.appendRow(['Roll No','Student','Course','Batch','Subject','Max Marks','Obtained','Grade','Entered By','Date']); data.records.forEach(r=>sh.appendRow([r.rollNo,r.studentName,r.course,r.batch,r.subject,r.maxMarks,r.obtainedMarks,r.grade,r.enteredBy,r.enteredAt||''])); return out({success:true}); } return out({success:false,error:'Unknown action'}); }catch(e){return out({success:false,error:e.toString()});} } function getSheet(ss,n){return ss.getSheetByName(n)||ss.insertSheet(n);} function out(o){return ContentService.createTextOutput(JSON.stringify(o)).setMimeType(ContentService.MimeType.JSON);} function doGet(){return out({status:'running'});}
